1.) Create file: sudo nano /etc/systemd/system/led-monitor.service
2.) Paste content below
3.) Enable service:
sudo systemctl daemon-reload
sudo systemctl enable led-monitor
sudo systemctl start led-monitor
==========================================================================


[Unit]
Description=LED System Activity Monitor
After=network.target
# Remove pigpiod dependency

[Service]
Type=simple
ExecStart=/usr/local/bin/led_monitor
Restart=always
RestartSec=5
Nice=19
CPUSchedulingPolicy=idle
IOSchedulingClass=idle
IOSchedulingPriority=7

User=root

MemoryLimit=50M
CPUQuota=5%

[Install]
WantedBy=multi-user.target
